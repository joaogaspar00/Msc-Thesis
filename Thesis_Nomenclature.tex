%
% The definitions can be placed anywhere in the document body
% and their order is sorted by <symbol> automatically when
% calling makeindex in the makefile
%
% The \glossary command has the following syntax:
%
% \glossary{entry}
%
% The \nomenclature command has the following syntax:
%
% \nomenclature[<prefix>]{<symbol>}{<description>}
%
% where <prefix> is used for fine tuning the sort order,
% <symbol> is the symbol to be described, and <description> is
% the actual description.
% The first letter in <prefix> indicates the group, while the next
% (if included) help sorting if the <symbol> is not a plain character
%
% ----------------------------------------------------------------------
% Roman symbols [r]
\nomenclature[ru]{$S_b$}{Blade Span}
\nomenclature[ru]{$N_b$}{Number of rotor's blades}
% ----------------------------------------------------------------------
% Greek symbols [g]
\nomenclature[g]{$\rho$}{Density}
\nomenclature[g]{$\alpha$}{Angle of attack}
\nomenclature[g]{$\psi$}{Azimutal Position}

\nomenclature[g]{$\delta\psi$}{Angle between two consecutive blades}

% ----------------------------------------------------------------------
% Subscripts [s]
\nomenclature[sx]{$x,y,z$}{Cartesian components}
\nomenclature[si]{$i,j,k$}{Computational indexes}
\nomenclature[s]{$\infty$}{Free-stream condition}
\nomenclature[sr]{ref}{Reference condition}
\nomenclature[sn]{$n$}{Normal component}

% ----------------------------------------------------------------------
% Supercripts [t]
\nomenclature[t]{T}{Transpose}
\nomenclature[t]{*}{Adjoint}
\nomenclature[t]{$i$}{Variable in the inertia referece frame}
\nomenclature[t]{$r$}{Variable in the rotor referece frame}
\nomenclature[t]{$b$}{Variable in the blade referece frame}
\nomenclature[t]{$e$}{Variable in the blade's element referece frame}
\nomenclature[t]{$a$}{Variable in the 2d aerodynamic referece frame}
